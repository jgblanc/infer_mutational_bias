'%!in%' <- function(x,y)!('%in%'(x,y))
HLA <- seq(28477797,33448354)
remove_HLA <- function(df) {
  bad <- subset(df, df$CHR == 6 & df$BP %in% HLA)
  bad_snps <- bad$SNP
  good <- df %>% filter(SNP  %!in% bad_snps)
  return(good)
}
